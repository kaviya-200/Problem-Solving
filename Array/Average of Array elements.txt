Average of Array Elements

import java.util.*;

public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int[] arr = new int[n];
        int sum=0;
        for (int i = 0; i < n; i++) { 
          arr[i]=sc.nextInt();
             sum+= arr[i]; 
        }
        int average = sum/n;
        System.out.println(average);
      
    }
}

5
10 20 30 40 50
Sample Output 0

30
