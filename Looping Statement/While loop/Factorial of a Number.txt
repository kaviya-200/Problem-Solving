 Factorial of a Number

import java.util.*;
public class FactWhile {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), fact = 1, i = 1;
        while (i <= n) {
            fact *= i;
            i++;
        }
        System.out.println(fact);
    }
}


Sample Input:
4
Sample Output:

24
