Prime Number Check

import java.util.*;
public class PrimeWhile {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), count = 0, i = 1;
        while (i <= n) {
            if (n % i == 0) count++;
            i++;
        }
        if (count == 2) System.out.println("Prime");
        else System.out.println("Not Prime");
    }
}


Sample Input:
10

Sample Output:
Not Prime
