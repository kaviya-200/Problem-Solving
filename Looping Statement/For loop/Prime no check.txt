Prime Number Check
import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int N=sc.nextInt();
        int c=0;
        for(int i=1;i<=N;i++){
            if(N%i==0){
                c++;
            }
        }
        if (c == 2) {
            System.out.println("Prime"+" ");
        } else {
            System.out.println("Not Prime");
        }
        
        
    }
}
Sample Input 0

10
Sample Output 0

Not Prime
Sample Input 1

2
Sample Output 1

Prime
