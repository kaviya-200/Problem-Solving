Factorial of a Number

import java.util.*;
public class FactFor {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt(), fact = 1;
        for (int i = 1; i <= n; i++) fact *= i;
        System.out.println(fact);
    }
}

Sample Input:
5
Sample Output:
120
