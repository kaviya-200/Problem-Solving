DECREASING STAR
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int size = sc.nextInt();
        int sum = 0;

        for (int i = size; i >= 1; i--) {
            for (int j = 1; j <= i; j++) {
                System.out.print("* ");
                sum++;
            }
            System.out.println();
        }
    }
}


Input:
4
Output:
* * * * 
* * * 
* * 
*
